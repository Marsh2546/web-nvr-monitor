# CCTV NVR Monitor - Google Sheets Daily Import Cron Job
# Run every day at 7:15 AM

# SPREADSHEET_ID=your_spreadsheet_id_here
# PG_HOST=localhost
# PG_PORT=5435
# PG_DATABASE=cctv_nvr

# Daily import at 7:15 AM
