## NVRStatus Condition

- ONU_STATUS
  - if ONU_STATUS == offline:
    NVR        = false
    HDD        = false
    CAMERA     = false
    LOGIN      = false
    NORMAL_VIEW = false
    END
- NVR_STATUS
  - if NVR_STATUS == offline:
    HDD        = false
    CAMERA     = false
    LOGIN      = false
    NORMAL_VIEW = false
    END
- HDD_STATUS
  - if ONU_STATUS == online AND NVR_STATUS == online:
    if HDD_STATUS == online:
        HDD = true
    else:
        HDD = false
  else:
      HDD = false
- LOGIN_STATUS
  - if NVR_STATUS == online:
      LOGIN = true
  else:
      LOGIN = false
- NORMAL_VIEW_STATUS
  - if NORMAL_VIEW == false:
      if SNAPSHOT_EXISTS == true:
        NORMAL_VIEW = true
      else:
        NORMAL_VIEW = false
